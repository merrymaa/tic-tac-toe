==== запрос на регистрацию нового пользователя:
curl -X POST http://localhost:5000/register \
-H "Content-Type: application/json" \
  -d '{
  "login": "user_6",
  "password": "pa3453454"
}'

echo -n "evg:pa3453454" | base64
